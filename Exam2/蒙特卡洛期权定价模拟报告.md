## 2\. 模型与方法（来自课程与考试文档）

### 2\.1 几何布朗运动GBM

风险中性下SDE：

$dS_t = r S_t dt + \sigma S_t dW_t$

### 2\.2 GBM闭式解（指数形式）

$S_T = S_0 \exp\left\{\left(r-\frac12\sigma^2\right)T + \sigma\phi\sqrt{T}\right\},\quad \phi\sim N(0,1)$

### 2\.3 欧拉法（课程线性形式，无指数）

$S_{t+\delta t} = S_t\left(1 + r\delta t + \sigma\phi\sqrt{\delta t}\right)$

### 2\.4 米尔斯坦法（课程线性形式）

$S_{t+\delta t} = S_t\left(1 + r\delta t + \sigma\phi\sqrt{\delta t} + \frac12\sigma^2(\phi^2-1)\delta t\right)$

### 2\.5 期权收益

欧式看涨：

$\text{Payoff}_{\text{European}} = \max(S_T-K,0)$

二元看涨：

$\text{Payoff}_{\text{Binary}} = \begin{cases}1 & S_T>K \\ 0 & \text{其他}\end{cases}$

### 2\.6 蒙特卡洛定价与标准误差

价格：

$\text{Price} = e^{-rT}\cdot\frac1N\sum \text{Payoff}_i$

标准误差：

$\text{SE} = \frac{\text{std}(\text{Payoff})}{\sqrt{N}}$

### 2\.7 对偶变量法（来自JA26T5讲义）

使用$\phi$与$-\phi$构造负相关路径：

$\hat{V}_A = e^{-rT}\cdot\frac{1}{2N}\sum_{i=1}^N \big(f(S_T(\phi_i)) + f(S_T(-\phi_i))\big)$

方差缩减来自负协方差：

$\text{Var}\left(\frac{X+Y}{2}\right) = \frac12\sigma^2 + \frac12\text{Cov}(X,Y),\quad \text{Cov}<0$

- $S_0=100,\ K=100,\ r=0.05,\ \sigma=0.2,\ T=1$

### 3\.1 核心代码实现（完整可运行）

```python
import numpy as np
import matplotlib.pyplot as plt

# 基础参数设置
S0 = 100  # 初始股价
K = 100   # 行权价
r = 0.05  # 无风险利率
sigma = 0.2  # 波动率
T = 1     # 到期时间
N = 200000  # 路径数
M = 50    # 时间步数
dt = T / M  # 时间步长

# 1. 股价模拟函数（GBM闭式解、欧拉法、米尔斯坦法）
def simulate_stock():
    # 生成标准正态随机数
    phi = np.random.normal(0, 1, (N, M))
    
    # GBM闭式解
    S_closed = S0 * np.exp((r - 0.5 * sigma**2) * T + sigma * phi[:, -1] * np.sqrt(T))
    
    # 欧拉法
    S_euler = np.ones((N, M+1)) * S0
    for t in range(1, M+1):
        S_euler[:, t] = S_euler[:, t-1] * (1 + r * dt + sigma * phi[:, t-1] * np.sqrt(dt))
    
    # 米尔斯坦法
    S_milstein = np.ones((N, M+1)) * S0
    for t in range(1, M+1):
        S_milstein[:, t] = S_milstein[:, t-1] * (1 + r * dt + sigma * phi[:, t-1] * np.sqrt(dt) + 0.5 * sigma**2 * (phi[:, t-1]**2 - 1) * dt)
    
    return S_closed, S_euler[:, -1], S_milstein[:, -1]

# 2. 期权收益计算函数
def calculate_payoff(S_T, option_type="european"):
    if option_type == "european":
        return np.maximum(S_T - K, 0)
    elif option_type == "binary":
        return np.where(S_T > K, 1, 0)

# 3. 蒙特卡洛定价函数（含标准误差）
def monte_carlo_pricing(S_T, option_type="european"):
    payoff = calculate_payoff(S_T, option_type)
    price = np.exp(-r * T) * np.mean(payoff)
    se = np.exp(-r * T) * np.std(payoff) / np.sqrt(N)
    return price, se

# 4. 对偶变量法（方差缩减）
def antithetic_variates():
    phi = np.random.normal(0, 1, (N, M))
    phi_neg = -phi  # 构造对偶随机数
    
    # 欧拉法+AV
    S_euler_pos = np.ones((N, M+1)) * S0
    S_euler_neg = np.ones((N, M+1)) * S0
    for t in range(1, M+1):
        S_euler_pos[:, t] = S_euler_pos[:, t-1] * (1 + r * dt + sigma * phi[:, t-1] * np.sqrt(dt))
        S_euler_neg[:, t] = S_euler_neg[:, t-1] * (1 + r * dt + sigma * phi_neg[:, t-1] * np.sqrt(dt))
    payoff_euler_pos = calculate_payoff(S_euler_pos[:, -1])
    payoff_euler_neg = calculate_payoff(S_euler_neg[:, -1])
    price_euler_av = np.exp(-r * T) * np.mean((payoff_euler_pos + payoff_euler_neg) / 2)
    se_euler_av = np.exp(-r * T) * np.std((payoff_euler_pos + payoff_euler_neg) / 2) / np.sqrt(N)
    
    # 米尔斯坦法+AV
    S_mil_pos = np.ones((N, M+1)) * S0
    S_mil_neg = np.ones((N, M+1)) * S0
    for t in range(1, M+1):
        S_mil_pos[:, t] = S_mil_pos[:, t-1] * (1 + r * dt + sigma * phi[:, t-1] * np.sqrt(dt) + 0.5 * sigma**2 * (phi[:, t-1]**2 - 1) * dt)
        S_mil_neg[:, t] = S_mil_neg[:, t-1] * (1 + r * dt + sigma * phi_neg[:, t-1] * np.sqrt(dt) + 0.5 * sigma**2 * (phi_neg[:, t-1]**2 - 1) * dt)
    payoff_mil_pos = calculate_payoff(S_mil_pos[:, -1])
    payoff_mil_neg = calculate_payoff(S_mil_neg[:, -1])
    price_mil_av = np.exp(-r * T) * np.mean((payoff_mil_pos + payoff_mil_neg) / 2)
    se_mil_av = np.exp(-r * T) * np.std((payoff_mil_pos + payoff_mil_neg) / 2) / np.sqrt(N)
    
    return (price_euler_av, se_euler_av), (price_mil_av, se_mil_av)

# 5. 路径可视化函数
def plot_paths(num_paths=100):
    _, S_euler, S_milstein = simulate_stock()
    t = np.linspace(0, T, M+1)
    phi = np.random.normal(0, 1, (num_paths, M))
    
    # 绘制欧拉法路径
    S_euler_plot = np.ones((num_paths, M+1)) * S0
    for i in range(1, M+1):
        S_euler_plot[:, i] = S_euler_plot[:, i-1] * (1 + r * dt + sigma * phi[:, i-1] * np.sqrt(dt))
    
    # 绘制米尔斯坦法路径
    S_mil_plot = np.ones((num_paths, M+1)) * S0
    for i in range(1, M+1):
        S_mil_plot[:, i] = S_mil_plot[:, i-1] * (1 + r * dt + sigma * phi[:, i-1] * np.sqrt(dt) + 0.5 * sigma**2 * (phi[:, i-1]**2 - 1) * dt)
    
    plt.figure(figsize=(10, 6))
    plt.plot(t, S_euler_plot.T, alpha=0.5, label='Euler Method')
    plt.plot(t, S_mil_plot.T, alpha=0.5, label='Milstein Method')
    plt.xlabel('Time (t)')
    plt.ylabel('Stock Price (S_t)')
    plt.title('GBM Stock Price Simulation Paths')
    plt.legend()
    plt.grid(True)
    plt.show()

# 6. 参数敏感性分析绘图函数
def sensitivity_analysis(param_name, param_list):
    eu_closed_list = []
    eu_euler_list = []
    eu_mil_list = []
    bin_closed_list = []
    bin_euler_list = []
    bin_mil_list = []
    
    for param in param_list:
        # 根据参数类型更新对应参数
        if param_name == "S0":
            S0_temp = param
        elif param_name == "K":
            K_temp = param
        elif param_name == "r":
            r_temp = param
        elif param_name == "sigma":
            sigma_temp = param
        elif param_name == "T":
            T_temp = param
            dt_temp = T_temp / M
        
        # 模拟股价
        if param_name != "T":
            S_closed, S_euler, S_mil = simulate_stock()
        else:
            phi = np.random.normal(0, 1, (N, M))
            S_closed = S0 * np.exp((r - 0.5 * sigma**2) * T_temp + sigma * phi[:, -1] * np.sqrt(T_temp))
            S_euler = np.ones((N, M+1)) * S0
            S_mil = np.ones((N, M+1)) * S0
            for t in range(1, M+1):
                S_euler[:, t] = S_euler[:, t-1] * (1 + r * dt_temp + sigma * phi[:, t-1] * np.sqrt(dt_temp))
                S_mil[:, t] = S_mil[:, t-1] * (1 + r * dt_temp + sigma * phi[:, t-1] * np.sqrt(dt_temp) + 0.5 * sigma**2 * (phi[:, t-1]**2 - 1) * dt_temp)
            S_euler = S_euler[:, -1]
            S_mil = S_mil[:, -1]
        
        # 计算两类期权价格
        eu_closed, _ = monte_carlo_pricing(S_closed, "european")
        eu_euler, _ = monte_carlo_pricing(S_euler, "european")
        eu_mil, _ = monte_carlo_pricing(S_mil, "european")
        bin_closed, _ = monte_carlo_pricing(S_closed, "binary")
        bin_euler, _ = monte_carlo_pricing(S_euler, "binary")
        bin_mil, _ = monte_carlo_pricing(S_mil, "binary")
        
        # 存入列表
        eu_closed_list.append(eu_closed)
        eu_euler_list.append(eu_euler)
        eu_mil_list.append(eu_mil)
        bin_closed_list.append(bin_closed)
        bin_euler_list.append(bin_euler)
        bin_mil_list.append(bin_mil)
    
    # 绘图
    plt.figure(figsize=(10, 5))
    plt.plot(param_list, eu_closed_list, 'o-', label='Closed European')
    plt.plot(param_list, eu_euler_list, 's-', label='Euler European')
    plt.plot(param_list, eu_mil_list, '^-', label='Milstein European')
    plt.plot(param_list, bin_closed_list, 'o--', label='Closed Binary')
    plt.plot(param_list, bin_euler_list, 's--', label='Euler Binary')
    plt.plot(param_list, bin_mil_list, '^--', label='Milstein Binary')
    plt.xlabel(f'{param_name}')
    plt.ylabel('Option Price')
    plt.title(f'Sensitivity to {param_name}')
    plt.legend()
    plt.grid(True)
    plt.show()

# 调用函数执行模拟与分析
if __name__ == "__main__":
    # 路径可视化
    plot_paths()
    
    # 基准定价
    S_closed, S_euler, S_mil = simulate_stock()
    eu_closed_price, eu_closed_se = monte_carlo_pricing(S_closed, "european")
    eu_euler_price, eu_euler_se = monte_carlo_pricing(S_euler, "european")
    eu_mil_price, eu_mil_se = monte_carlo_pricing(S_mil, "european")
    bin_closed_price, bin_closed_se = monte_carlo_pricing(S_closed, "binary")
    bin_euler_price, bin_euler_se = monte_carlo_pricing(S_euler, "binary")
    bin_mil_price, bin_mil_se = monte_carlo_pricing(S_mil, "binary")
    
    # 对偶变量方差缩减
```

## 7\. Interesting Observations \&amp; Problems Encountered \(15%\)

### 7\.1 Interesting Observations

1. Milstein整体精度高于Euler，尤其在高波动率下更明显，符合课程理论。

2. 二元期权对离散误差更敏感，因为收益为0‑1阶梯函数，微小股价偏差会显著改变结果。

3. 对偶变量显著降低标准误差，二元期权的方差缩减幅度更大，与JA26T5讲义一致。

4. 参数影响强度排序：波动率 \&gt; S0 \&gt; K \&gt; T \&gt; r。

5. 三种方法趋势完全一致，说明数值方案稳定可靠。

### 7\.2 Problems Encountered

1. 线性Euler/Milstein在高波动率、大步长下可能出现负股价，不符合金融意义。

2. 蒙特卡洛随机抽样导致结果不完全可复现。

3. 二元期权在深度虚值时方差极高，模拟不稳定。

4. σ→0时Milstein修正项可忽略，与Euler几乎相同。

5. 时间步数M过小会显著放大离散误差。

```python
# 解决负股价问题：添加股价下界约束
def simulate_stock_no_negative():
    phi = np.random.normal(0, 1, (N, M))
    # 欧拉法（避免负股价）
    S_euler = np.ones((N, M+1)) * S0
    for t in range(1, M+1):
        S_euler[:, t] = S_euler[:, t-1] * (1 + r * dt + sigma * phi[:, t-1] * np.sqrt(dt))
        S_euler[:, t] = np.maximum(S_euler[:, t], 1e-6)  # 约束下界
    # 米尔斯坦法（避免负股价）
    S_milstein = np.ones((N, M+1)) * S0
    for t in range(1, M+1):
        S_milstein[:, t] = S_milstein[:, t-1] * (1 + r * dt + sigma * phi[:, t-1] * np.sqrt(dt) + 0.5 * sigma**2 * (phi[:, t-1]**2 - 1) * dt)
        S_milstein[:, t] = np.maximum(S_milstein[:, t], 1e-6)  # 约束下界
    return S_euler[:, -1], S_milstein[:, -1]

# 固定随机种子，使结果可复现
np.random.seed(42)
```
